# 冒泡排序

从第一个数起和相邻元素进行比较，如果前一个数比后一个数大，则交换，一趟排序过后从后往前确定一个数

外层i(比较次数)：len - 1 >= i > =1内层j（拿到比较的元素）：0 < = j < i arr[j] 、arr[j+1]比较

function bubbleSort(arr){
    let len = arr.length;
    let temp;
    // 表示有交换元素的标志位
    let flag = false
    //
    for(let i = len - 1 ; i > 0; i--){
        for(let j = 0; j < i; j++){
            if(arr[j] > arr[j+1]){
                temp = arr[j+1];
                arr[j+1] = arr[j];
                arr[j] = temp;
                flag = true
            }
        }
        //如果没有发生一次交换 就退出 提升性能
        if(!flag){
            break;
        }else{
            flag = false
        }
    }

}

# 选择排序

从数组中选择一个最小的数。假定最小索引，从剩余元素中找到实际最小索引，交换，就确定了一个元素

外层i（每次假定的最小索引） 0<= i <=len-2 内层j（每次选择的范围）：i+1<= j < len

function selectSort(arr){
    let len = arr.length;
    let temp;
    for(let i = 0; i < len - 1; i++){
        // 假定最小索引
        let minIndex = i
        for(let j = i + 1; j < len ; j++){
            if(arr[minIndex] > arr[j]) {
                minIndex = j
            }
        }
        //如果没有在后面区间中找到minIndex 证明我们假定正确 不用交换 提高性能
        if(minIndex !== i){
            temp = arr[i]
            arr[i] = arr[minIndex];
            arr[minIndex] = temp
        }
    }


}

# 插入排序

假定数组一部分是有序的，一部分是无序的，从无序中找一个元素插入到有序中的合适位置

从1号元素开始往前面插，把它与左边第一个元素比较，如果左边第一个元素比它大，则继续与左边第二个元素比较下去，直到遇到不比它大的元素，然后插到这个元素的右边

外层i（每次待插入的元素arr[i]）内层j（寻找可插入位置 insertIndex为i-1 前面元素比待插入的大 则arr[insertIndex+1] = insertVal）

​

function insertSort(arr){
    let len = arr.length;
    for(let i = 1 ; i<len; i++){
        //一开始有序表中只有数组的第一项 从1~n-1中挑出来插到有序表中
        //保存待插入的项
        let insertVal = arr[i];
        //假设插在待插入项左边的位置
        let insertIndex = i - 1;
        //待插入的项小于他要插入的位置的话 移位
        while(insertIndex >= 0 && insertVal < arr[insertIndex]){
            //进入循环要移动插入的位置
            arr[insertIndex + 1] = arr[insertIndex]
            insertIndex --;
        }
        arr[insertIndex+1] = insertVal
    }
}

# 希尔排序

是插入排序的一种变种，交换不相邻的元素以达到数组的局部排序

选择分组gap（第一次选取len/2，每两个为一组），gap每次除以2，gap>=1

外层i（gap <= i < len ）内层j（ gap<= j <= i ; j-=gap）

function shellSort2(arr){
    let len = arr.length;
    let gap = Math.floor(len / 2);
    for(;gap > 0 ;gap=Math.floor(gap/2)){
    //    从第gap元素逐个对其所在的组进行直接插入排序
    /*
        一开始gap为5 待排序的为arr[5] temp保存
        0-5比较 arr[5] < arr[0] 先不交换 arr[0]大的就赋给arr[5]
        给arr[5]一次性找到插入的位置
    */
        for(let i = gap; i < len ; i++){
            let j = i;
            let temp = arr[j]
            //一次性移动到最前的
            //要移动的如果比他前面的小 理应移动
            while(j-gap>=0 && temp < arr[j-gap]){
                    //移动arr[j]的位置交给比她大的那位
                    arr[j] = arr[j-gap];
                    j-=gap;
                }
            //    退出循环找到位置了
            arr[j] = temp;
        }
    }
    return arr;
}

# 归并排序

用到的思想有：递归，分治、双指针；定义一个主方法、sort方法进行递归、merge方法j进行合并

对半拆分数组 直到low-mid mid+1 - high 拆为两个单个元素数组 将他们合并

合并数组同时排序完成 双指针：同时从两个待合并数组开始移动，较小的放到辅助数组中，直至一个数组遍历完，将剩下元素也加入辅助数组中，最后替换到到大数组中

// 递归写法
function main(){
    let arr = [8,4,5,7,1,3,6,2];
    let temp = new Array(arr.length).fill(0);
    mergeSort(arr,0, arr.length-1, temp);
    //console.log(arr)
}

// 8 4 5 7 -> 8 4 -> 都进不去if循环 -> 下标01开始合并 -> 5 7  -> 进不去if循环 下标23开始合并 
// 1 3 6 2
function mergeSort(arr,left,right,temp){
    if(left < right){
        let mid = Math.floor((left + right) / 2);
    //    向左递归分解
    //    剩两个就会进行到合并的阶段
        mergeSort(arr, left, mid, temp)
    //    向右递归分解
        mergeSort(arr,mid+1, right, temp)
    //    合并
        merge(arr,left,mid, right,temp);
    }
}

//合并的方法
//arr:排序原始数组
//left左边有序序列的初始索引
//mid 数组的中间索引
//right右边索引
//temp做中转的数组
function merge(arr, left, mid, right, temp){
    let i = left;
    let j = mid + 1;
    let t = 0 ; //指向temp数组的当前索引
    //1、左右两边有序数组按规则填充感到temp数组 直到左右两边的有序序列有一方处理完毕为止
    while(i <= mid && j <= right){
        //左边有序序列当前元素小于等于右边有序序列的当前元素
        //赋值 后移
        if(arr[i] <= arr[j]){
            temp[t++] = arr[i++];
        }else {
            temp[t++] = arr[j++];
        }
    }
    //2、把有剩余数据的一边的数据依次全部填充到temp
    //判断谁有剩
    while( i <= mid){
    //左边有剩余就全部填充到temp
        temp[t++] = arr[i++]
    }
    while(j <= right){
        //右边有剩余就全部填充到temp
        temp[t++] = arr[j]
    }
    //3、将temp数组的元素拷贝到arr 并不是每次都拷贝所有
    t = 0;
    let tempLeft = left;
    //最后一次才是全部拷贝
    while(tempLeft <= right){
        arr[tempLeft] = temp[t++]
        tempLeft ++
    }
}

# 快速排序

选取最左边的元素作为pivot，分割成两个子序列，小的都放pivot左边，大的都放pivot右边，一趟下来，pivot所处的位置就确定了

再对分割的两个子数组进行递归重复操作。

function quickSort(arr,left,right){
    let l = left, r= right;
    //找到标志位 假设是最左边的那位
    let pivot = arr[left];
    if(l < r){//避免死循环
        while(l <r){
            while(l < r && arr[r] >= pivot){
                r --; //从右往左找到一个小于pivot的值
            }
            if(l < r){
                //将找到的小于pivot的值赋给l处
                arr[l] = arr[r];
                ++l;
            }
            while( l<r && arr[l] < pivot){
                l++; //从左往右找到一个大于pivot的值
            }
            if(l < r){
                //将找到的小于pivot的值赋给r处
                arr[r] = arr[l]
                r--;
            }
        }
        arr[l] = pivot;
        //一趟过后l和r相等 左边都比pivot小或等于 右边同理
        quickSort(arr,left, l-1);
        quickSort(arr,r+1, right);
    }
}

# 堆排序

堆排序是一种选择排序，堆顶是最大值是大顶堆，堆顶是最小值是小顶堆，左节点是2*i + 1,右节点是2* i + 2

将数组构建成一个大顶堆，堆顶是最大值 ，次将堆顶元素放在最后，和最后一个元素交换减小调整堆的长度

初始化堆：从堆的第一个非叶子结点（下标为len /2 - 1）开始调整，找寻当前、左、右三节点的最大值

function heapSort(arr){
    let temp = 0
    //形成一个大顶堆后就要把第一个元素换到最后 arr就能减少了
    //初始化堆
    for(let i = (Math.floor(arr.length/2) -1); i >=0; i--){
        adjustHeap(arr,i,arr.length)
    }
    // 每次调整范围减小
    for(let j = arr.length - 1 ; j>0 ; j--){
        temp = arr[j];
        arr[j] = arr[0]
        arr[0] = temp
        //交换后向下进行筛选
        adjustHeap(arr,0 ,j)
    }
}
//将一个数组（二叉树）调整成大顶堆
//参数说明：arr数组 i表示非叶子节点在二叉树顺寻存储数组的索引
//len表示调整范围的长度 len逐渐减少
//功能：对i指向对应的非叶子节点的树进行调整
function adjustHeap(arr, i, len){
    let temp = arr[i]; //取出当前元素的值
    //k是i指向节点的左子节点
    //for循环后将这个局部子树放在了传入i的那个位置，即最顶部
    for(let k = (i*2+1); k < len; k = k * 2 +1){
        //相当于做了一个 根、左、右三方比较
        //根变为最大 并将i最后指向左右节点中更大的那个
        if(k+1 <len && arr[k] < arr[k+1]){
            //说明左子节点的值小于右子节点
            //k指向左右结点中较大的节点
            k++
        }
        if(arr[k] > temp){
            arr[i] = arr[k];//把较大的值赋给当前节点 为根
            i = k; //i指向k
        }else{
            break;
        }

    }
    arr[i] = temp; //将temp放到调整后的位置

}

# 桶排序

在最大值和最小值之间进行瓜分，例如分为10个区间，再对每个桶中的元素装入对应的区间中，对每个桶进行排序，之后进行合并汇总,q我们取桶数量等于原始数列的元素数量，

区间跨度: d = (max-min)/ ( bucketNum - 1)

获取装入哪个桶的公式 Math.floor((arr[i] - min) / d )

function bucketSort(arr) {
  let len = arr.length;
  let min = arr[0], max = arr[0];
  // 找出数组中的最大值和最小值
  for (let i = 1; i < len; i++) {
    if (max < arr[i]) {
      max = arr[i]
    }
    if (arr[i] < min) {
      min = arr[i]
    }
  }
  // 最大值最小值之差
  let d = max - min;
  // 桶的个数 = 数组的长度
  let bucketNum = len;
  // 一个二维数组
  let bucketList = new Array(bucketNum).fill(0)
  for(let i = 0; i < bucketNum; i++){
    bucketList[i] = new Array()
  }
  for (let i = 0; i < len; i++) {
    // 求出放进哪个桶里 相当于超出最小值的部分除以区间跨度看放在哪个桶里
    let num = Math.floor(((arr[i] - min) * (bucketNum - 1)) / d)
    bucketList[num].push(arr[i])
    
  }
  // 这样出来的桶 最后一个桶会只能放最大值
  for (let bucket of bucketList) {
    bucket.sort()
  }

  let sortArr = new Array(len);
  let index = 0;
  for (let bucket of bucketList) {
    for (let item of bucket) {
      sortArr[index++] = item
    }
  }
  return sortArr
}

# 基数排序

按个位数的大小对数据进行排序，接着按十位，对某位数进行排序的时候，都是用桶来排序的，某位数的取值范围是0-9，需要10个桶

比如个位都是3的先放一个桶里，放完后取出来覆盖原数组，把10 个桶清空，然后对十位、百位都重复操作

function radioSort(arr){
    let len = arr.length;
    let max = arr[0]
    // 找到最大值
    for(let i= 1; i < len; i++){
        if(arr[i] > max){
            max = arr[i]
        }
    }
    let max_bit = 0
    // 求出最大值的位数
    while (Math.floor(max) / 10 > 0){
        max_bit ++;
        max = Math.floor(max / 10)
    }
    //10个桶
    //每个桶定义为一个数组
    let bucket = new Array(10).fill(0).map(()=>[])
    for(let i = 0, n = 1; i < max_bit; i++){
        for(let j = 0 ; j < arr.length; j++){
            // 拿到对应位数 空缺就为0 拿到个位
            let radio = Math.floor(arr[j] / n) % 10;
            // 放入对应的桶中
            bucket[radio].push(arr[j])
        }
        
        let k =0;
        for(let j =0 ; j < 10; j++){
            for(let item of bucket[j]){
                arr[k++] = item
            }
            bucket[j] = []
        }
        n *= 10;
    }
}

# 计数排序

适合最大值和最小值的差值不是很大的排序，因为数组下标值为整数，所以计数排序对小数搞不定，

构造一个临时数组进行计数（长度为max-min + 1，对最小值到最大值之间可能出现的值都要预留出一个下标），对原数组中的值进行计数，temp[i] = m 表示值为i的元素在原数组中出现m次 然后再从temp中依次取出

function countSort(arr){
  let len = arr.length;
  let min = arr[0], max = arr[0]
  // 找出原数组的最大值 最小值
  for(let i =1; i < len ; i++){
    if(arr[i] > max){
      max = arr[i]
    }
    if(arr[i] < min){
      min = arr[i]
    }
  }
  // 临时数组的长度
  let d = max - min + 1
  // 构造临时数组 统计出现次数
  let temp = new Array(d).fill(0)
  for(let i = 0; i < len; i++){
    temp[arr[i] - min] ++;
  }
  let k = 0;
  for(let i = 0; i < d; i++){
    // 下标i＋min就还原出原数组的值
    // 循环出现的次数
    for(let j = temp[i]; j > 0; j--){
      arr[k++] = i + min
    }
  }
  return arr
}

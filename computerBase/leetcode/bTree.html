# 不同的二叉搜索树

本题是二叉树和动态规划的结合，动态规划难点在于定义状态和定义递推式，其中dp[ i ] [ j ]一般定义为数组前 i 个元素咋咋咋满足求和为 j 或者达到 j 的状态，递推式一般需要推导

var numTrees = function (n) {
  // 需要推导的dp
  // 遍历一个有序序列 以i为根节点 1到i-1则为其左子树 i+1到n则为其右子树
  // 设G(n)即dp(n) 为序列长度n的互相不同的二叉搜索树有多少种
  // 设F(i,n)为序列长度为n 以i为根节点的二叉搜索树有多少种
  // G(n)就等于F(i,n) i从1到n的累加
  // F(i,n) = G(i-1) * G(n - i)
  // 例如F(3,7) 以3为根节点左边序列就是1-2 右边序列4-7 总共多少种就是左右两边种类的乘积
  const dp = new Array(n + 1).fill(0);
  dp[0] = 1 // 空 有1种
  dp[1] = 1 // 长度为1也只有1种
  for (let i = 2; i <= n; i++) {
    for(let j = 1; j <=i ;j++){
      // 从以1为根节点到序列的最后为根节点
      dp[i] += dp[j-1]*dp[i-j]
    }
  }
  return dp[n]
};

# 二叉搜索树的后序遍历

二叉搜索树：满足左子树所有节点值比当前节点值小，右子树所有节点值都比当前节点值大

后序遍历最后一个位置是根节点，从头开始首先遍历的是左边，所有理应比最后根节点的值小，当找到比根节点大的值时，即已开始遍历右子树

通过该规律， 递归检查整棵树的后序遍历和其子树的后序遍历

var verifyPostorder = function(postorder) {
    //如果是后序排列 且每个子树都是后序排列 一个后序排列的最后一项是根节点
    //
    return recur(postorder, 0, postorder.length-1)

    function recur(postorder, i, j){
        // i在最前 j在最后
        //只剩一个节点
        if(i >= j) return true;
        let p = i;
        while(postorder[p] < postorder[j]) p++
        let m = p;
        while(postorder[p] > postorder[j]) p++
        return p == j && recur(postorder,i,m-1) && recur(postorder,m, j-1)
    }
};

var flatten = function(root){
    if(!root) return
    let stack = []
    stack.push(root);
    let pre = null;
    while(stack.length){
        // 弹出先弹出根节点->弹出根节点的左节点->弹出根节点的右节点
        let cur = stack.pop();
        if(pre !== null){
            pre.right = cur;
            pre.left = null;
        }
        // 保存右节点 保证右节点不丢失
        if(cur.right !== null){
            stack.push(cur.right)
        }
        // 保存左节点
        if(cur.left !== null){
            stack.push(cur.left)
        }
        pre = cur;
    }
}

# 二叉搜索树与双向链表

二叉搜索树的中序遍历是一个有序序列，对其进行中序遍历，要达到：head指向最左下角的元素

提供一个pre指向上一次遍历到节点，pre的右指针指向当前节点，当前节点的左指针指向pre，遍历结束后pre指向最右下角的节点

var treeToDoublyList = function(root) {
    //  二叉搜索树的中序遍历是一个有序序列 在中序遍历的时候进行构建
    if(!root){
        return null;
    }
    let pre = null;
    // pre其实指向了当前遍历到的节点
    // head 最后是要指向最左下角的节点的
    // pre遍历到最后指向了最右下角的节点
    let head = null;
    dfs(root)
    pre.right = head
    head.left = pre
    return head;
    // 中序遍历
    function dfs(root){
      if(!root){
          return null;
      }
      // 中序遍历
      dfs(root.left)
      // 难的地方
      // pre的右指针指向当前
      // 当前的左指针指向pre
      if(pre){
          pre.right = root;
      }else{
          // 只会在pre为null时进入 也就是最左下角的系节点
          head = root
      }
      root.left = pre;
      pre = root;
      dfs(root.right)
    }

};

# 二叉树的最近公共祖先

二叉树的题目基本上就是递归了，本题要返回某个具体结点

最近公共祖先有两种情况：pq在某个祖先两侧，p是祖先，q在p的子树中；q是祖先，p在p的子树中

终止条件：找到p或者q，或者遇到叶节点

递归工作：拿到左节点、右节点返回的结果，几位

返回值：根据left和right的情况返回

var lowestCommonAncestor = function(root, p, q) {
    // 设最近公共祖先为root pq可能位于root的异侧 或者 p和q有一者为祖先
    // 递归遍历树 遇到越过叶子节点、p、q则返回
    if(root == null || root == q || root == p) return root;
    // 当前节点向左向右遍历 分别有left和right两个返回值
    let left = lowestCommonAncestor(root.left, p , q);
    let right = lowestCommonAncestor(root.right, p ,q);
    // left right 都为null 则都没找到pq
    // left 为null right不为null pq其中之一在right里面  right就是p/q 或者pq在right的子树里面
    // left 和 right 都不会null  那么pq存在于当前节点的异侧 即找到了目标值
    // 返回的left 和 right 有两种结果
    if(left == null) return right;
    if(right == null) return left;
    return root;

};

var isCousins = function (root, x, y) {
  let Xdepth = 0, Xparent = null, Xfind = false,
    Ydepth = 0, Yparent = null, Yfind = false;
  //广度优先搜索
  //创建辅助队列
  let q = [[root, 0]]
  find(q, null, 0)
  function find(n, parent, depth) {
    if (n.val === x) {
      Xparent = parent
      Xfind = true
      Xdepth = depth
    }
    else if (n.val === y) {
      Yparent = parent
      Yfind = true
      Ydepth = depth
    }
  }
  while (q.length) {
    let [node, depth] = q.shift()
    if (Xfind && Yfind) {
      break;
    }
    if (node.left) {
      q.push([node.left, depth + 1])
      find(node.left, node, depth + 1)
    }
    if (node.right) {
      q.push([node.right, depth + 1])
      find(node.right, node, depth + 1)
    }
  }
  return (Xdepth === Ydepth) && (Xparent !== Yparent)
};

# 树的子结构

两步走：选定A的某个节点和B树开始比较；从A的根节点开始，每个节点找一遍

var isSubStructure = function(A, B) {
    // 如果A或者B本身是空树 则false
    if(!A || !B){
        return false
    }
    // 判断在A的结构或者子结构中有没有跟B相等的
    return isSameNode(A,B) || isSubStructure(A.left, B)||isSubStructure(A.right, B)
    //实现一个函数比较两个节点是否相等
    // 传入两个节点相等则继续递归
    function isSameNode(A, B){
        if(!B){//B遍历完了 表示B的结构都能找到相等的
            return true
        }
        if(!A){ //A遍历完了但是B还没遍历完
            return false
        }
        if(A.val != B.val){
            return false
        }
        //节点相等递归子树
        return isSameNode(A.left, B.left) && isSameNode(A.right, B.right)

    }
};

# 二叉树的深度

var maxDepth = function(root){
    if(!root) return 0;
    // 遍历到叶子节点的时候返回上一层的值为1
    return Math.max(maxDepth(root.left),maxDepth(root.right)) + 1
}

# 无重复的最长字串

滑动窗口和哈希表的结合，用哈希表存储已在窗口中的元素且不能重复。依次移动右边界，遇到和哈希表里重复的，则移动窗口左边界，统计最长字串。

var longthOfLongestSubstring = function (s){
    // 用哈希表做滑动窗口 无重复 想到集合类型
    let set = new Set();

    let left = 0, right = 0;
    let len = s.length;
    let ans = 0;
    while(left < len){
        while(right < len && !set.has(s[right])){
            set.add(s[right]);
            right ++;
        }
        // 退出循环后指向重复位
        ans = Math.max(ans, right - left);
        // 滑动窗口内重复了移动左指针 缩小窗口
        set.delete(s[left])
        left ++
    }
    return ans;
}

# 有效的数独

用哈希表判断是否在一行、一列、一个小宫格内存在重复元素，有重复的即返回false

var isValidSudoku = function(board) {
    let rows = [], cols = [], boxes = []
    for(let i = 0; i < 9 ; i++){
        rows[i] = [] // 有9行
        cols[i] = [] // 有9列
        boxes[i] = [] // 有9个宫格
    }
    //分别创建三个哈希表存储已遍历到的行、列、宫格内的元素
    //
    //循环每一行每一列 取出值填进去 检查有无重复项
    for(let i = 0 ;i < 9 ; i++){
        for(let j = 0; j < 9 ; j++){
            let temp = board[i][j];
            //检查不是空的
            if( temp != '.'){
                //检查行 如果不具有这个元素
                if( !rows[i].includes(temp)){
                    rows[i].push(temp)
                }else{
                    return false
                }
                //检查列
                if( !cols[j].includes(temp)){
                    cols[j].push(temp)
                }else{
                    return false
                }
                // 拿到当前行当前列在哪个盒子里
                let boxIndex = Math.floor( i / 3) * 3  + Math.floor( j / 3)
                if( !boxes[boxIndex].includes(temp)){
                    boxes[boxIndex].push(temp)
                }else{
                    return false
                }
            }
        }
    }
    return true
};

# 最长连续子序列

HashSet去重，因为只要找到最长的序列即可，该序列肯定不会重复，如果当前遍历的数字减一的数存在于哈希表中，则跳过，因为它不是起点。

function longestConsecutive(nums){
  // 朴素做法：
  let maxLen = 0
  let len = nums.length
  for(let i = 0; i < len; i ++){
    // 找到一个x 然后以x为起点 在数组中找x+1 x+2
    let target = nums[i], temp = 1, flag = true
    for(let j = target + 1 ; j < target + len; j ++){
      if(flag){
        flag = false
        for(let num of nums){
            if(j == num){
              // 在原数组找到了 退出
              flag = true;
              temp ++
              break;
            }
          }
      }else{
          break;
      }
    }
    maxLen = Math.max(temp, maxLen)
  }
  return maxLen
}
var longestConsecutive = function(nums) {
  // 哈希表 先去重 取出一个元素 判断它-1的元素是否在集合中 没有就当作连续子序列的第一个开始寻找长度
  const set = new Set();
  for(let item of nums){
    set.add(item) ; //集合去重
  }
  let maxL = 0;
  for(let num of nums){
    if(!set.has(num-1)){
      //证明找到了连续子序列的第一个元素
      let cur = num
      let curL = 1;
      while(set.has(cur + 1)){
        curL += 1;
        cur += 1;
      }
      maxL = Math.max(maxL, curL)
    }
  }
  return maxL
};

# 环形链表II

快慢指针法

哈希表法

var detectCycle = function(head) {
    // 快慢指针
    // fast一次走2步 slow一次走一步
    // 总步数：fast = 2 * slow
    // 快指针比slow多走了nb fast = slow + n * b
    // 解得 fast = 2nb slow = nb
    // 所以第一次相遇 slow走了n个环的长度 要走到入口处要a + nb 所以还需走a步
    // 第二次 让fast从head处一次走一步 直至相遇 二者指向的就是环入口处
    let slow = head, fast = head;
    do{
      if(!fast || !fast.next) return null;
      slow = slow.next;
      fast = fast.next.next
    }while(slow !== fast);
    fast = head;
    while(slow !== fast){
      slow = slow.next;
      fast = fast.next;
    }
    return slow;
};

# LRU缓存机制

哈希表查找快，但是无序，链表中删除需要访问前驱节点，单向链表不行，满足插入、删除插入快，要用双向链表，所以把二者结合。建立一个双向链表，按照被使用的顺序，每次插入、删除都会维护这个双向链表的顺序，而又一个哈希表存储在双向链表上的结点值

class DoubleLinkedListNode{
    constructor(key, val){
        this.key = key;
        this.val = val;
        this.pre = null;
        this.next = null;
    }
}
class LRUCache{
    constructor(capacity){
        this.capacity = capacity;
        this.map = new Map(); // 存储键值对
        this.dummyHead = new DoubledLinkedListNode(null, null);
        this.dummyTail = new DoubledLinkedListNode(null, null);
        this.dummyHead.next = this.dummyTail;
        this.dummyTail.pre = this.dummyHead;
    }
    removeFromList(node){
        node.next.pre = node.pre;
        node.pre.next = node.next;
        node.next = null;
        node.pre = null;
        return node;
    }
    addToHead(node){
        const head = this.dummyHead.next;
        node.next = head;
        head.pre = node;
        this.dummyHead.next = node;
        node.pre = this.dummyHead;
    }
    get(key){
        if(this.map.has(key)){
            // 如果已存在
            let node = this.map.get(key);
            this.addToHead(node);
            return node.val;

            return -1;
        }

        if(this.map.has(key)){
            // 已存在 更新值

# 实现前缀树

顶层对应一个哈希表，26个键，字符串中的一个字母对应着一个键，一个单词对应一个嵌套对象

var Trie = function(){
    this.map = {}; // 顶层的map
    // apple对应：{ a: { p : { p : { l : { e : {}, isEnd: true } } } } }
}
Trie.prototype.insert = function(word){
    let map = this.map
    for(let c of word){
        if(!map[c]){
            map[c] = {}
        }
        map = map[c]; //递进 即已经存储了一个字母
    }
    map.isEnd = true; //加一个字段表示该单词已经结束
}
Trie.prototype.search = function(word) {
  let res = this.startsWith(word)
  return res !== undefined && res.isEnd !== undefined;
};
Trie.prototype.startsWith = function(prefix) {
  let map =  this.map
  for(let ch of prefix){
    if(!map[ch]){
      return false
    }
    map = map[ch]
  }
  return map
};

# 找到字符串中所有字母异位词

滑动窗口中是一些字符，需要用哈希表记录窗口内包含p中的字符出现的次数，当某个字符出现的次数和p字符串中相同字符出现的次数一致时，有效字符数加1

var findAnagrams = function(s, p){
    let res = [];
    const need = new Map();
    for(let s of p){
        if(need.has(s)){
            need.set(s, need.get(s)+ 1)
        }else{
            need.set(s, 1)
        }
    }
    // 定义窗口
    let window = new Map();
    let left = 0, right = 0, valid = 0;
    while(right< s.length){
        const c = s[right];
        right ++;
        if(need.has(c)){
            window.set(c, window.has(c) ? window.get(c) + 1 : 1);
            if(window.get(c) === need.get(c)){
                valid ++;
            }
        }
        // 长度一致 窗口满了
        if(right - left == p.length){
            // 无重复字符数和哈希表中统计的一致
            if(valid == need.size){
                res.push(left); //起始位置
            }
            const d = s[left];
            left ++;
            if(need.has(d)){
                if(window.get(d) == need.get(d)){
                    valid --
                }
                window.set(d, window.get(d) - 1)
            }
        }
    }
    return res;
}

# 连续的子数组和

前缀和+哈希表，判断当前位置的前缀和是否满足条件

前缀和一般是用来确定一个区间，有两个下标的前缀和除以K余数一样，那么就说明他们中间的范围是k的倍数

var checkSubarraySum = function(nums, k){
    let map = new Map();
    map.set(0, -1); //初始化 如果有一个数的前缀和正好为k的倍数，那么就有可能是从头开始
    let len = nums.length;
    let sum = 0
    for(let i = 0; i < len; i++){
        sum += nums[i]
        let y = sum % k;
        if(map.has(y)){
            let preIndex = map.get(y)
            if(i - preIndex >= 2){//没有包括preIndex指向的元素
                return true
            }
        }else{
            map.set(y, i)
        }
    }
    return false;
}

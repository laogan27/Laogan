# III

动态规划和二叉树的结合

var rob = function(root) {
  // 动态规划存储在哈希表中，记录每个节点在符合条件下能偷取到的最高金额
  // 需要两个哈希表 一个存储当前结点被选中后的最高金额 一个存储当前结点没被选中的最高金额
  let selected = new Map(), unselected = new Map();
  dfs(root)
  return Math.max(selected.get(root), unselected.get(root))
  function dfs(node){
    if(!node) return;
    dfs(node.left);
    dfs(node.right);
    // 当前结点被选中 则左右结点都不能被选中
    selected.set(node, node.val + (unselected.get(node.left) || 0) + (unselected.get(node.right) || 0));
    // 当前结点没被选中 则左右结点可以不被选中 也可以被选中
    unselected.set(node, Math.max(selected.get(node.left) || 0, unselected.get(node.left) || 0) + Math.max(selected.get(node.right) || 0, unselected.get(node.right) || 0));


  }
};

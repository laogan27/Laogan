# 路径总和

# III

用普通dfs需要从不同节点开始计算路径


var pathSum = function(root, targetSum){
    // 从根节点出发
    let res = rootSum(root, targetNum)
    // 分别递归计算从左右节点出发
    res += rootSum(root.left, targetNum);
    res += rootSum(root.right, targetNum);
    return res;
    var rootSum = function(root, targetNum){
        let res = 0;
        if(!root) return res;
        let val = root.val
        if(val == targetNum) res++;
        res += rootSum(root.left, targetNum - val);
        res += rootSum(root.right ,targetNum - val);
        return res;
    }
}

上述方法每次res只能加1，并且需要从每个节点出发寻找一条路径等于目标和的

利用前缀和，记录根节点出发的到节点p1,p2,p3....的路径和

var pathSum = function(root, targetNum){
    let prefix = new Map();
    // 初始化 和为0的为1
    prefix.set(0, 1);
    let res = dfs(root, prefix, 0, targetNum)
    return res;
    function dfs(node, prefix, cur, targetNum){
        if(!node) return 0;
        let res = 0;
        let val = res.val;
        // 遍历到当前节点的总和
        cur += val;
        res += prefix.get(cur - targetNum) || 0;
        prefix.set(cur, (prefix.get(cur) || 0 )+ 1);
        // 看找不着得到目标和前去当前和的前缀和
    	res += dfs(node.left, prefix, cur, targetSum)
    	res += dfs(node.right, prefix, cur, targetSum)
        // 遍历完该节点了 就应该把当前节点前缀和删去
    	prefix.set(cur, (prefix.get(cur) || 0) - 1)
    	return res;
    }
}

call() 方法使用一个指定的 this 值和单独给出的一个或多个参数来调用一个函数。

Function.prototype.myCall = function (context){
    // 保存this指向
    context = context ? context: window;
    // 原函数调用this 保存原函数
    context.func = this;
    // 收集传入的参数
    let arr = new Array()
    for(let i = 1; i < arguments.length; i++){
        arr.push(arguments[i])
    }
    context.func(...arr)
    delete context.func;
    
}

apply() 方法调用一个具有给定this值的函数，以及以一个数组（或类数组对象 (opens new window)）的形式提供的参数。

Function.prototype.myApply = function (context, arr){
    context = context || window;
    // 用symbol值唯一标识 保证不重复
    // 保存原函数
    const key = Symbol()
    context[key] = this;
    if(!arr){
        context[key]()
    }else{
        if(Array.isArray(arr)){
            context[key](...arr)
        }else{
            throw Error('There must be array！')
        }
    }
}

# Bind

MDN

bind() 方法创建一个新的函数，在 bind() 被调用时，这个新函数的 this 被指定为 bind() 的第一个参数，而其余参数将作为新函数的参数，供调用时使用。

Function.prototype.myBind2 = function (context, ...args){
    // 保存原函数
    let self = this;
    context = context || window;
    // 构造所要返回的函数
    let newFn = function (...rest){
        // let newArgs = Array.prototype.slice.call(arguments);
        if(this instanceof newFn){
            return self.apply(this, args.concat(rest))
        }else{
            return self.apply(context, args.concat(rest))
        }
    }
    // 如果是构造函数 实际上就是在new Self原函数 但是传入的this是newFn的实例
    // new出来的实例应该instanceof 原函数、新函数都为true 所以得维护原型
    // 没加上维护原型 new出来的实例是newFn的实例 但原型链上没有self
    let Temp = function (){};
    Temp.prototype = self.prototype;
    newFn.prototype = new Temp();
    // newFn.protoType = new self()
    // this -> newFn -> temp -> Temp -> self.protoType
    // new出来的实例，修改之后原型指向Temp的实例
    // Temp实例的原型指向原函数的原型
    return newFn;
    
}

# 函数柯里化

定长参数

function curry(fn){
    // 获取原函数的参数长度
    const argLen = fn.length;
    const presetArgs = [].slice.call(arguments, 1)
    return function(){
        const restArgs = [].slice.call(arguments);
        const allArgs = [...presetArgs, ...restArgs];
        if(allArgs.length >= argLen){
            return fn.apply(this, allArgs);
        }else{
            return curry.call(null, fn, ...allArgs)
        }
    }
}

# New

MDN

new 运算符创建一个用户定义的对象类型的实例或具有构造函数的内置对象的实例。

new 关键字会进行如下的操作：

  1. 创建一个空的简单JavaScript对象（即{}）；
  2. 为步骤1新创建的对象添加属性__ proto __，将该属性链接至构造函数的原型对象 ；
  3. 将步骤1新创建的对象作为this的上下文 ；
  4. 如果该函数没有返回对象，则返回this。
function myNew(constructor, ...args){
    if(typeof constructor !== 'function'){
        throw Error('There must be function!')
    }
    let obj = {}
    obj.__proto__ = constructor.prototype;
    let res = constructor.apply(obj, args);
    // return null 也返回新创建出来的对象
  	return typeof res == 'object' ? res || obj : obj;;
}

function deepCopy(obj, map = new WeakMap()){
  if(typeof obj !== 'object' && obj == null) return obj;
  if(obj instanceof Date) return new Date(obj);
  if(obj instanceof RegExp) return new RegExp(obj);
  // 如果拷贝过该对象
  if(map.has(obj))  {
    return map.get(obj);
  }
  //let newObj = new obj.constructor();
  let newObj = obj instanceof Array ? [] : {}
  // weakMap只能存储对象键
  map.set(obj, newObj)
  for(let key in obj){
    if(obj.hasOwnProperty(key)){
      newObj[key] = deepCopy(obj[key], map)
    }
  }
  return newObj
}

Object.myCreate = function (proto, propertiesObject){

# Object.assign

MDN

Object.assign() 方法用于将所有可枚举和自有的属性从一个或多个源对象分配到目标对象。它将返回目标对象。

/*
	参数：
	target: 目标对象，不能是null，也不能为undefined
	sourceObj：可选，null和undefined会被忽略，原始数据类型会转化成包装对象，
	注意，只有字符串的包装对象才可能有自身可枚举属性。
*/
Object.myAssign = function(target, sourceObj){
    if(target === null || target === undefined){
        throw new Error('There')
    }
    // 以目标对象创建一个新对象 引用是一样的
    let newObj = Object(target)
    for(let i = 1; i < arguments.length; i++){
        let source = arguments[i];
        if(source !== null && source !== undefined){
            //继承属性和不可枚举的属性是不可拷贝的
            //源对象上的属性会覆盖 后面的源对象会覆盖前面的源对象 的同名属性
           for(let key in source){
            	if(source.hasOwnProperty(key)){
                	newObj[key] = source[key]
            	}
        	}
        }
    }
    return newObj
}

# Object.is

MDN

Object.is() 方法判断两个值是否为同一个值 (opens new window)。

与== (en-US) (opens new window) 运算不同。 == 运算符在判断相等前对两边的变量(如果它们不是同一类型) 进行强制转换 (这种行为的结果会将 "" == false 判断为 true), 而 Object.is不会强制转换两边的值。

与=== (en-US) (opens new window) 运算也不相同。 === 运算符 (也包括 == 运算符) 将数字 -0 和 +0 视为相等 ，而将Number.NaN (opens new window) 与NaN (opens new window)视为不相等.

Object.is = function(x, y){
    if(x === y){
        // 不是+0 或 0 或 -0 就会直接返回true
        return x !== 0 || 1 / x === 1 / y;
    }else{
        // NaN === NaN 为false
        // 只要NaN !== NaN为true 不是NaN就返回false
        return x !== x && y !== y;
    }
}

    }

}

# reduce

MDN

reduce() 方法对数组中的每个元素执行一个由您提供的reducer函数(升序执行)，将其结果汇总为单个返回值。

callback

执行数组中每个值 (如果没有提供 initialValue则第一个值除外)的函数，包含四个参数：

  • accumulator

    累计器累计回调的返回值; 它是上一次调用回调时返回的累积值，或initialValue（见于下方）。

  • currentValue

    数组中正在处理的元素。

  • index 可选

    数组中正在处理的当前元素的索引。 如果提供了initialValue，则起始索引号为0，否则从索引1起始。

  • array可选

    调用reduce()的数组

initialValue可选

作为第一次调用 callback函数时的第一个参数的值。 如果没有提供初始值，则将使用数组中的第一个元素。 在没有初始值的空数组上调用 reduce 将报错。

WARNING

回调函数第一次执行时，accumulator 和currentValue的取值有两种情况：如果调用reduce()时提供了initialValue，accumulator取值为initialValue，currentValue取数组中的第一个值；如果没有提供 initialValue，那么accumulator取数组中的第一个值，currentValue取数组中的第二个值。

如果数组为空且没有提供initialValue，会抛出TypeError (opens new window) 。如果数组仅有一个元素（无论位置如何）并且没有提供initialValue， 或者有提供initialValue但是数组为空，那么此唯一值将被返回并且callback不会被执行。

Array.prototype.reduce = function(cb, initialVal){
    if(this === null) throw new TypeError()
    if(typeof cb !== 'function') throw new TypeError()
    var o = Object(this)
    var len = o.length >>> 0;
    var k = 0;
    var value = initialVal;
    // 如果没传初值
    if(initialVal === undefined){
        // 如果是空数组 不满足k in o 
        while(k < len && !(k in o)){
            k++
        }
        // 如果是空数组且没提供初值则报错
        if(k >= len){
            throw new TypeError('Reduce of empty array with no inital value')
        }
        // 如果没提供初值 则初值为第一个元素
        value = o[k++]
    }
    while(k < len){
       if (k in o) {
          value = callback(value, o[k], k, o);
        }
        k++;
      }
      return value;
    }
    
}

function parseParams(url){
    const paramsStr = /.+\?(.+)$/.exec(url)[1]; //拿到后面的query参数
    const paramsArr = paramsStr.split('&');
    let paramsObj = {};
    paramsArr.forEach(param => {
        if(/=/.test(param)){
            ley [key, val] = param.split('=');
            key = decodeURIComponent(key);
            val = decodeURIComponent(val);
            val = /^\d+$/.test(val) ? parseFloat(val) : val;
            if(paramsObj.hasOwnProperty(key)){
                paramsObj[key] = [].concat(paramsObj[key], val);
            }else{
                paramsObj[key] = val;
            }
        }else{
            // 处理没有value的参数
            paramsObj[param] = true;
        }
    })
    return paramsObj;
}

        }

# 原型模式

将公共的方法和属性添加到构造函数的prototype属性上，共有的属性的共享问题会造成同步修改

每个函数都会创建一个prototype属性，即调用构造函数创建的对象的原型，定义在原型上的属性和方法是被所有实例共享的

实例的[[prototype]]属性指向原型对象，原型对象上的construcor指向构造函数，构造函数和实例之间没有联系

实例对象上也能访问到constructor属性，因为原型链查找

  • isPrototypeOf(obj)：当传入的对象内部的[[prototype]]属性执行调用该函数的对象时，返回true

  • getPrototypeof(obj)：返回传入对象内部的[[prototype]]属性值，获取对象的原型

  • Object.create(proto)：创建一个新对象，同时为其指定原型

  • hasOwnProperty()：确定某个属性是在实例上还是在原型对象上

  • property in obj：in操作符可以通过对象访问到指定属性时就返回true

  • for..in：可以通过对象访问且可以枚举的属性都会返回，包括实例属性和原型属性

  • Object.keys()：返回对象自身所有可枚举属性名称的字符串数组

  • Object.getOwnPropertyNames()：列出对象自身的属性，无论是否可枚举，如constructor

属性枚举顺序：Object.getOwnPropertyNames()，Obejct.getOwnPropertySymbols()，Object.assign()的枚举顺序是确定性的：先以升序枚举数值键，再以插入顺序枚举字符串和符号键

  • 初始需求：Promise存在三种状态，pending、fulfilled、rejected，类似状态机；成功时，从pending转为fulfilled，且不可逆，只能从pending转为fulfilled，失败同理

    调用promise构造函数，传入一个执行函数（executor），该函数也需要两个参数（resolve，reject）两个函数在内部定义，如果运行在 excutor 函数中的业务执行成功了，会调用 resolve 函数；如果执行失败了，则调用 reject 函数

    Promise的then方法，接受两个回调函数参数：onFulfilled、onRejected，当Promise成功时执行成功回调，失败时执行失败回调

  • 异步需求：内部resolve方法的调用一般需要等待外部用户的异步操作以后执行，外部有异步操作，promise还没被resolve变为完成状态就调用then方法了，所以如果在pending状态下调用then方法，需要将传入的回调函数进行保存，等resolve或者reject的时候再取出调用

  • 执行器函数中的resolve，可以传入一个Promise，失败态的promise就是失败，成功态的promise就是成功

  • 链式调用的需求：then方法返回一个promise

  • then方法中的回调是异步调用的需求：加上setTimeout模拟

  • then方法的注意点：then方法返回一个Promise，供之后继续then

    • 返回一个值，then返回的Promise是成功态，返回值是成功值，传递给下个then的成功回调
    • 不返回任何值，then返回的Promise是成功态，成功值是undefined
    • 抛出一个错误，then返回的Pormise是失败态，接收失败原因，传给下个then的失败回调
    • 返回一个已是成功态的Promise，then返回的Promise是成功态，并且将那个Promise的成功值变成返回的成功值
    • 返回一个已是失败态的Promise，then返回的Promise是失败态，并且将那个Promise的失败值变成返回的失败原因
    • 返回一个new Promise，即未定状态，then返回的Promise也是未定的，并且和那个Promise的终态保持一致

  • Promise.resolve不一定返回一个成功态的promise

class Promise{
    constructor(executor){
        // 初始状态
        this.promiseState = 'pending';
        // promise返回值
        this.promiseResult = undefined
        // 在内部定义resolve，reject方法
        // 两个方法主要在作用是 改变promise的状态以及保存要传给then方法或者catch方法的值
        // resolve是一个函数 接受一个要传给then的值
        // 保存成功回调
        this.onResolveedCallbacks = [];
        // 保存失败回调
        this.onRejectedCallbacks = [];
        let reject = reason => {
            if(this.promiseState === 'pending'){
                this.promiseState = 'rejected';
                this.promiseResult = reason;
                this.onRejectedCallbacks.forEach(fn=>{
                    fn()
                })
            }
        }
        let resolve = value => {
            // resolve也可以是promise
            if(value instanceof MyPromise){
                // value是Promise的执行结果a then中成功回调为resolve(a) 改变状态
                // return value.then(v => {resolve(v)}, e =>{reject(e)})
                // 此语句为上语句的简便写法
                return value.then(resolve, reject)
            }
            if(this.promiseState === 'pending'){
                this.promiseState = 'fulfilled';
                this.promiseResult = value;
                this.onResolvedCallbacks.forEach(fn=>{
                    fn()
                })
            }
        }
        // 调用执行函数
        try{
        	executor(resolve, reject);
        }catch(e){
            reject(e);
        }
    }
    // 这两个函数是用户传的，仅需在调用时传入对应的值
    // 还需判断用户传入的函数参数的返回值 以便链式调用
    // then方法返回一个Promise
    then(onFulfilled, onRejected){
        // 确保then方法接收的这两个参数时函数
        //如果第一个参数不是函数，则会在内部被替换为 (x) => x，
        //如果第二个参数不是函数，则会在内部被替换为一个 "Thrower" 函数 (it throws an error it received as argument)。
        onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : val => val;
        onRejected = typeof onRejected === 'function' ? onRejected : reason =>{ throw reason }
        function resolvePromise(res, thenPromise, resolve, reject){
            try{
                //拿到then方法回调的返回值 执行了！
                //判断其返回值以指示then方法返回的promise的状态和promise返回值
                if(res === thenPromise){
                    throw new Error('不能返回自身')
                }
                let called;
                // 如果是对象类型 判断then是不是函数 即thenable
                if((typeof res === 'object' && res !== null) || typeof res === 'function'){
                    try{
                        let then = res.then;
                        if(typeof then === 'function'){
                            then.call(res, v =>{
                                if(called) return;
                                called = true;
                                resolvePromise(v, thenPromise, resolve, reject)
                            }, err=>{
                                if(called) return;
                                called = true;
                                resolvePromise(err, thenPromise, resolve, reject)
                            })
                        }
                    }catch(e){
                       if(called) return;
                        called = true;
                        reject(e)
                    }
                }else{
                    // 是普通值直接resolve
                    resolve(x)
                }
            }catch(e){
                reject(e) //thenPromise为失败的promise
            }
    	}
        let thenPromise = new Promise((resolve, reject) =>{
            //在这里面以resolve、reject决定状态
              if(this.promiseState === 'fulfilled'){
                setTimeout(()=>{
                  let res = onFulfilled(this.promiseResult)
                  resolvePromise(res, thenPromise, resolve, reject)
                }, 0)
              }
              else if(this.promiseState === 'rejected'){
                setTimeout(()=>{
                  let res = onRejected(this.promiseResult)
                  resolvePromise(res, thenPromise, resolve, reject)
                }, 0)
              }
              else if(this.promiseState === 'pending'){
                this.onResolvedCallbacks.push(()=>{
                  setTimeout(()=>{
                    let res = onFulfilled(this.promiseResult)
                    resolvePromise(res, thenPromise, resolve, reject)
                  }, 0)
                })
                this.onRejectedCallbacks.push(()=>{
                  setTimeout(()=>{
                    let res = onRejected(this.promiseResult)
                    resolvePromise(res, thenPromise, resolve, reject)
                  }, 0)
                })
              }
       })
       return thenPromise
    }
    // 静态方法实现
    static resolve(data){
		return new MyPromise((resolve, reject)=>{
            resolve(data)
        })
    }
    static reject(err){
		return new MyPromise((resolve, reject)=>{
            reject(err)
        })
    }
    static all(promiseArrs){
		let len = promiseArrs.length;
        let res = []
        return new MyPromise((resolve, reject)=>{
            for(let i = 0; i < len; i++){
                Promise.resolve(promiseArrs[i]).then(data=>{
                    res.push(data)
                    if( i == len - 1){
                        resolve(res)
                    }
                }, reject)
            }
        })
    }
    static race(promiseArrs){
		return new MyPromise((resolve, reject)=>{
            for(let i = 0; i < promiseArrs.length; i++){
                Promise.resolve(promiseArrs[i]).then(resolve, reject)
            }
        })
    }
    static finally(cb){
        // 采用上一次的结果 即上一次点调用的promise
        // 传入一个回调执行
		return this.then((value)=>{
            return Promise.resolve(cb()).then(()=>value)
        }, (reason)=>{
            return Promise.resolve(cb()).then(()=> {throw reason})
        })
    }
}

# 实现Promise.allSettled

function allSettleds(promises){
    return new Promise((resolve, reject) => {
        let count = 0;
        let res = [];
        let len = promises.length;
        if(len === 0){
            return resolve([])
        }
        promises.forEach((p, i) => {
            Promise.resolve(p).then((res) => {
                count += 1;
                res[i] = {
                    status: 'fulfilled',
                    value: res;
                }
            })
                .catch((err)=>{
                count += 1;
                res[i] = {
                    status: "rejected",
                    reason: err
                }
            })
                .finally(()=>{
                if(count === len){
                    resolve(res);
                }
            })
        })
    })
}

# 发布订阅模式

在观察者和订阅者之间多了一个事件通道作为中间过渡，目标对象发生了变化，不直接通知观察者，而是通过事件通道去通知观察者做出更新，因此观察者也只需要订阅事件通道的消息即可。

class Pubsub{
    constructor(){
        this.events = {}
    }
    // 观察者订阅 on监听
    on(eventType, fn){
        if(this.events.eventType){
            this.events.eventType.push(fn)
        }else{
            this.events.eventType = [fn]
        }
    }
    // 订阅者通知观察者做出更新
    emit(eventType, ...args){
        if(this.events.eventType){
            for(let fn of this.events.eventType){
                fn(...args)
            }
        }
    }
    off(eventType, offFn){
        if(this.events[eventType]){
            let index = this.events[eventType].findIndex((fn)=>{
                return fn === offFn
            })
            this.events[eventType].splice(index, 1);
            if(!this.events[eventType]){
                delete this.events[eventType];
            }
        }
    }
}

# Axios核心原理

# 基础使用原理

axios是一个基于promise的http库，还是属于XMLHttpRequest，因此需要一个ajax，或者基于http

基本使用方式：axios(config)、axios.method(url, data, config)

class Axios{
    constructor(){
        
    }
    request(config){
        return new Promise((resolve)=>{
            const { url = '', method = 'get' , data = {}} = config;
            // 发送ajax请求
            const xhr = new XMLHttpRequest();
            xhr.open(method, url, true);
            xhr.onload = function(){
                console.log(xhr.responseText);
                resolve(xhr.responseText)
            }
            xhr.send(data)
        })
	}
}
function createAxios(){
    let axios = new Axios();
    let req = axios.request.bind(axios);
    return req;
}
const methodArr = ['get', 'post','put', 'delete'];
methodArr.forEach(method => {
    Axios.prototype[method] = function(){
        console.log('执行'+method +"方法");
        if(['get', 'delete', 'head', 'option'].includes(method)){
            return this.request({
                method: method,
                url: arguments[0],
                ...arguments[1] || {}
            })
        }else{
            return this.request({
                method: method,
                url: arguments[0],
                data: arguments[1] || {},
                ...arguments[2] || {}
            })
        }
    }
})

# 拦截器原理

基本使用：axios.interceptors.request.use(success, error) / axios.interceptors.response.use(success, error)

基本原理：请求拦截放在chain数组前面，响应拦截放在chain数组最后，中间api请求，初始化chain，也需要将interceptors 挂载到request对象上

class InterceptorsManage{
    constructor(){
        this.handlers = [];
    }
    use(fullfied, rejected){
        this.handlers.push({
            fullfied,
            rejected
        })
        return this.handlers.length - 1;//返回一个id可以取消拦截
    }
}
class Axios(){
    this.interceptors = {
        request: new InterceptorsManage(),
        response: new InterceptorsManage()
    }
    request(config){
        let chain = [this.sendAjax.bind(this，config), undefined] // 成对出现
        this.interceptors.request.handlers.forEach(interceptor => {
            chain.unshift(interceptor.fullfied, interceptor.rejected)
        })
        this.interceptors.response.handlers.forEach(interceptor => {
            chain.push(interceptor.fullfied, interceptor.rejected)
        })
        let promise = Promise.resolve(config);
        while(chain.length > 0){
            // 每次成对弹出作为then的成功回调和失败回调
            promise = promise.then(chain.unshift(), chain.unshift());
        }
        return promise;
    }
    sendAjax(config){
    	return new Promise(resolve => {
        const {url = '', method = 'get', data = {}} = config;
        // 发送ajax请求
        console.log(config);
        const xhr = new XMLHttpRequest();
        xhr.open(method, url, true);
        xhr.onload = function() {
            console.log(xhr.responseText)
            resolve(xhr.responseText);
        };
        xhr.send(data);
    	})
	}
}

# 取消请求原理

通调用cancelToken构造函数创建一个实例对象，并创建一个cancel函数

调用cancel函数，取消请求

基本使用：

const CancelToken = axios.CancelToken;
// 在这里调用source方法返回一个对象
const source = CancelToken.source();
let cancel
axios.get(url, {cancelToken: source.token } )
source.cancel('xxxx')

class CancelToken{
    // new一个cancel实例的时候 新建了promise 将resolve交给cancel函数
    constructor(executor){
        this.message = ""
        var resolvePromise;
        this.promise = new Promise((resolve, reject)=>{
            resolvePromise = resolve;
        })
        var token = this;
        executor(function wrap(message){
            if(token.message){
                
            }
            token.message = message;
            // 关键操作
            resolvePromise(message)
        })
    }
}
CancelToken.source = function(){
    var cancel ;
    // new 一个实例
    var token = new CancelToken(function executor(c){
        // c就是上面wrap传递出来给cancel
        cancel = c;
    })
    return {
        cancel,// cancel是用户调用的 接收message错误信息赋给token实例
        token;
    }
}
// 在sendAjax方法中 如果config传入cancelToken了
sendAjax(config){
    if(config.cancelToken){
// promise的resolve传到了cancel函数了，当用户调用了cancel的时候resolve（message）才进入then，才可以abort
        config.cancelToken.promise.then((cancelMessage)=>{
            if(!xhr){
                return;
            }
            xhr.abort();
            reject(cancelMessage);
            xhr = null;
        })
    }
}
